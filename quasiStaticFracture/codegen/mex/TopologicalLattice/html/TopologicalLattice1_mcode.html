1function TopologicalLattice(domain_wall, notch_length, wall_distance, label)
  2
  3%{
  4
  5Lattice fracturing simulation.
  6Quasi static uniaxial pulling.
  7
  8domain_wall = true / false
  9notch_length is the length of the notch
 10wall_distance is the domain wall distance
 11label is a unique label for file output
 12
 13
 14%}
 15
 16
 17%% coder extrinsic
 18
 19% declear external matlab function for coder
 20coder.extrinsic('sprintf')
 21coder.extrinsic('initial')
 22coder.extrinsic('breaking_threshold')
 23coder.extrinsic('break_bonds')
 24coder.extrinsic('figure_putput')
 25coder.extrinsic('stress')
 26coder.extrinsic('notch')
 27
 28% pre assign matrices with variable shapes
 29
 30% X =zeros(72000,2);  
 31coder.varsize('X', [72000, 2], [1 0]);
 32X0 = zeros(7200,2);
 33coder.varsize('X0', [72000, 2], [1 0]);
 34B = zeros(72000,2);
 35coder.varsize('B', [72000, 2], [1 0]);
 36S1 = zeros(7200,1);
 37coder.varsize('S1', [7200, 1], [1 0]);
 38S2 = zeros(7200,1);
 39coder.varsize('S2', [7200, 1], [1 0]);
 40S = zeros(1,72000);
 41coder.varsize('S', [1, 72000], [0 1]);
 42W = 0.0;
 43H = 0.0;
 44B_broken = B;
 45coder.varsize('B_broken', [72000, 2], [1 0]);
 46B_broken0 = B;
 47coder.varsize('B_broken0', [72000, 2], [1 0]);
 48G = zeros(72000,1);
 49coder.varsize('G', [72000, 1], [1 0]);
 50snapshot = 0;
 51stress_force_yy = 0.0;
 52avalanche = 0;
 53fileID = 0;
 54t=0;
 55
 56%% massive production of fracture.
 57
 58%% The first step is to sample around 20 configs for the two cases.
 59%  Thresholds will first be 0.05 +/- 0.01 for this moment.
 60
 61%% Management of DIR
 62
 63% cd '/Users/leyou/Box Sync/MATLAB/TL/Fracture'
 64
 65if domain_wall
 66    file_avalanche = sprintf('data_notch/avalanche_wall_notch%d_walldistance_%dlabel%d.txt',notch_length,wall_distance,label);
 67    file_stress = sprintf('data_notch/stress_wall_notch%d_walldistance_%d_label%d.txt',notch_length,wall_distance,label);
 68else
 69    file_avalanche = sprintf('data_notch/avalanche_control_notch%d_label%d.txt',notch_length,label);
 70    file_stress = sprintf('data_notch/stress_control_notch%d_label%d.txt',notch_length,label);
 71end
 72
 73%% Main flow
 74
 75% 1. initialize the lattice. 
 76
 77show_plot = false;
 78B_broken = zeros(0,2);
 79B_broken0 = zeros(0,2);
 80% domain_wall = true;
 81[X,B,S1,S2,S,W,H]=initial(50,80,show_plot,domain_wall,wall_distance);
 82
 83X0 = X;
 84
 85
 86% 1.5 Add notch (vertical, in the center)
 87
 88    [B] = notch(X,B,W,H,notch_length);
 89    
 90% 2. Assign random breaking threshold
 91
 92    [G] = breaking_threshold(0.05,0.005,B,label);
 93
 94% 3. Loop of strain
 95
 96for strain = [0.00:0.001:0.2]
 97    
 98%     strain
 99    snapshot = snapshot + 1;
100
101% 4. Energy minimization
102
103    flag_relaxation = true;
104    B_broken0 = B_broken;
105    while flag_relaxation 
106        for i = [1:numel(S2)]
107            j = S2(i);
108            X(j,2) = X0(j,2) + X0(j,2) * strain;
109        end
110        [X] = relaxation(X,X0,B,S1,S2,W);
111
112
113% 5. Breaking of bonds
114    
115        [B,B_broken,G,flag_relaxation]=break_bonds(X,X0,B,B_broken,S,G,W);
116    end
117    
118
119% 6. (Optional) Figure output
120    
121    if show_plot
122        figure_output()
123    end
124
125% 7. Store results in files
126    [stress_force_yy] = stress(X,X0,B,S2,S,W);
127    avalanche = size(B_broken,1) - size(B_broken0,1);
128    fileID = fopen(file_stress,'a');
129    fprintf(fileID,'%16.8f %16.8f\n',strain,stress_force_yy);
130    fclose(fileID);
131    if avalanche > 0
132        fileID = fopen(file_avalanche,'a');
133        fprintf(fileID,'%16.8f %16.8f\n',strain,avalanche);
134        fclose(fileID);
135    end
136    
137end
138
139
140end
